1 /**
  2  * @name $.ui.simplicityPagination
  3  * @namespace Pagination widget for simplicityDiscoverySearch.
  4  * <p>
  5  * Pagination widget bound to the <code>simplicitySearchResponse</code> event.
  6  * <p>
  7  * The current page can be bound either via an <code>input</code> element or
  8  * directly to the <code>simplicityState</code>.
  9  *
 10  * @example
 11  *   // Bound directly to simplicityState
 12  *   <div id="pagination"></div>
 13  *   <script type="text/javascript">
 14  *     $('#pagination').simplicityPagination();
 15  *   </script>
 16  *
 17  * @example
 18  *   // Bound to an input
 19  *   <input id="page" name="page" />
 20  *   <div id="pagination"></div>
 21  *   <script type="text/javascript">
 22  *     $('#page').simplicityInputs();
 23  *     $('#pagination').simplicityPagination({
 24  *       input: '#page'
 25  *     });
 26  *   </script>
 27  */
 28 (function ($, window) {
 29   $.widget("ui.simplicityPagination", $.ui.simplicityWidget, {
 30     /**
 31      * Widget options.
 32      *
 33      * <dl>
 34      *   <dt>stateElement</dt>
 35      *   <dd>
 36      *     The location of the simplicityState widget. Defaults to <code>'body'</code>.
 37      *   </dd>
 38      *   <dt>searchElement</dt>
 39      *   <dd>
 40      *     The location of the simplicityDiscoverySearch widget. Defaults to <code>'body'</code>.
 41      *   </dd>
 42      *   <dt>pageParam</dt>
 43      *   <dd>
 44      *     Used when binding directly to the <code>simplicityState</code> and determines which
 45      *     field the state holds the current page number. Defaults to <code>'page'</code>.
 46      *   </dd>
 47      *   <dt>input</dt>
 48      *   <dd>
 49      *     When set binds the current page to an <code>input</code> element instead of directly
 50      *     to the state via <code>pageParam</code>. Defaults to <code>''</code>.
 51      *   </dd>
 52      *   <dt>applyClass</dt>
 53      *   <dd>
 54      *     Classes to apply to the Next, Prev and Page elements. Defaults to <code>ui-corner-all</code>.
 55      *   </dd>
 56      *   <dt>scrollTopSelector</dt>
 57      *   <dd>
 58      *     Determines the jQuery selector to apply <code>$(window).scrollTop(0)</code> when
 59      *     a page changes. Defaults to <code>window</code>.
 60      *   </dd>
 61      *   <dt>scrollTopPosition</dt>
 62      *   <dd>
 63      *     Determines the scroll top position to use for <code>$(window).scrollTop(0)</code>. Defaults to <code>0</code>.
 64      *   </dd>
 65      *   <dt>debug</dt>
 66      *   <dd>
 67      *     Enable logging of key events to <code>console.log</code>. Defaults to <code>false</code>.
 68      *   </dd>
 69      * </dl>
 70      * @name $.ui.simplicityPagination.options
 71      */
 72     options : {
 73       stateElement: 'body',
 74       searchElement: 'body',
 75       pageParam: 'page',
 76       input: '',
 77       applyClass: 'ui-corner-all',
 78       scrollTopSelector: window,
 79       scrollTopPosition: 0,
 80       debug: false
 81     },
 82     _create : function () {
 83       this
 84         ._addClass('ui-simplicity-pagination')
 85         ._bind(this.options.searchElement, 'simplicitySearchResponse', this._searchResponseHandler)
 86         ._bind(this.options.stateElement, 'simplicityStateReset', this._stateResetHandler);
 87     },
 88     /**
 89      * Event handler for the <code>simplicitySearchResponse</code> event. Recreates
 90      * the pagination widget to reflect the current search response.
 91      *
 92      * @name $.ui.simplicityPagination._searchResponseHandler
 93      * @function
 94      * @private
 95      */
 96     _searchResponseHandler: function (evt, searchResponse) {
 97       var target = $('<div/>');
 98       if (searchResponse) {
 99         var discoveryResponse = searchResponse._discovery || {};
100         var resultSet = discoveryResponse.response || {};
101         var numItems = resultSet.totalSize;
102         var itemsPerPage = resultSet.pageSize;
103         var currentPage = resultSet.startIndex / itemsPerPage;
104         try {
105           this._ignoreCallback = true;
106           target.pagination(
107             numItems,
108             $.extend({
109               current_page: currentPage,
110               items_per_page: itemsPerPage,
111               callback: $.proxy(this._paginationCallback, this)
112             }, this.options)
113           );
114           target.find('a')
115             .addClass('ui-state-default')
116             .addClass(this.options.applyClass);
117           target.find('span.current').each($.proxy(function (idx, ele) {
118             var span = $(ele);
119             if (span.hasClass("prev") || span.hasClass("next")) {
120               span
121                 .addClass('ui-state-disabled ui-priority-primary')
122                 .addClass(this.options.applyClass);
123             } else {
124               span
125                 .addClass('ui-state-active ui-priority-primary')
126                 .addClass(this.options.applyClass);
127             }
128           }, this));
129           this.element.find('div.pagination').remove();
130           this.element.append(target.find('div.pagination'));
131         } finally {
132           this._ignoreCallback = false;
133         }
134       }
135     },
136     /**
137      * Callback for the upstream pagination widget that gets called when a page change action
138      * has been taken.
139      *
140      * @name $.ui.simplicityPagination._paginationCallback
141      * @function
142      * @private
143      */
144     _paginationCallback: function (page) {
145       if (!this._ignoreCallback) {
146         if (this.options.scrollTopSelector !== '') {
147           $(this.options.scrollTopSelector).scrollTop(this.options.scrollTopPosition);
148         }
149         if (this.options.input !== '') {
150           // Store the page in an input
151           $(this.options.input).val(page + 1);
152           $(this.options.input).change();
153         } else {
154           // Store the page directly in the search state
155           var state = $(this.options.stateElement).simplicityState('state');
156           state[this.options.pageParam] = String(page + 1);
157           $(this.options.stateElement).simplicityState('state', state);
158         }
159         if (this.options.searchElement !== '') {
160           if (false === $(this.options.searchElement).simplicityDiscoverySearch('option', 'searchOnStateChange')) {
161             $(this.options.searchElement).simplicityDiscoverySearch('search');
162           }
163         }
164       }
165       return false;
166     },
167     /**
168      * Event handler for the <code>simplicityStateReset</code> event. Resets the current page
169      * if the <code>pageParam</code> option is set.
170      *
171      * @name $.ui.simplicityPagination._stateResetHandler
172      * @function
173      * @private
174      */
175     _stateResetHandler: function (evt, state) {
176       if (this.options.input === '') {
177         if (this.options.debug) {
178           console.log('simplicityPagination: Resetting state parameter', name, 'for', this.element);
179         }
180         delete state[this.options.pageParam];
181       }
182     }
183   });
184 }(jQuery, window));
