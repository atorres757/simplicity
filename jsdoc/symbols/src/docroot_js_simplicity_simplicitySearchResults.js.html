1 /**
  2  * @name $.ui.simplicitySearchResults
  3  * @namespace Widget that displays the search results.
  4  *
  5  * This widget listens to the <code>simplicitySearchResponse</code> event and takes
  6  * the <code>resultsHtml</code> field from the search response, injecting it directly
  7  * into the page.
  8  * <p>
  9  * If your search controller does not inject the search results as HTML into it's response
 10  * then consider using <code>simplicityRenderParamsSearchResults</code> instead.
 11  */
 12 (function ($) {
 13   $.widget("ui.simplicitySearchResults", $.ui.simplicityWidget, {
 14     /**
 15      * Widget options.
 16      *
 17      * <dl>
 18      *   <dt>searchElement</dt>
 19      *   <dd>
 20      *     The simplicityDiscoverySearch widget that this widget binds it's events to. Defaults to <code>'body'</code>.
 21      *   </dd>
 22      * </dl>
 23      * @name $.ui.simplicitySearchResults.options
 24      */
 25     options: {
 26       searchElement: 'body'
 27     },
 28     _create : function () {
 29       this
 30         ._addClass('ui-simplicity-search-results')
 31         ._bind(this.options.searchElement, 'simplicitySearchResponse', this._searchResponseHandler);
 32     },
 33     /**
 34      * Event handler for the <code>simplicitySearchResponse</code> event. Expects the given
 35      * search response object to contain a <code>resultsHtml</code> field and splices it
 36      * directly into the page.
 37      *
 38      * @name $.ui.simplicitySearchResults._searchResponseHandler
 39      * @function
 40      * @private
 41      */
 42     _searchResponseHandler: function (evt, searchResponse) {
 43       var resultsHtml = searchResponse.resultsHtml || '';
 44       if (searchResponse.error) {
 45         var error = $('<div class="ui-state-error-text"/>');
 46         error.text('[' + searchResponse.status + '] ' + searchResponse.statusText + ((searchResponse.message) ? ", " + searchResponse.message : ""));
 47         this.element.append(error);
 48       } else {
 49         this.element.html(resultsHtml);
 50       }
 51     }
 52   });
 53 }(jQuery));
