1 /**
  2  * @name $.ui.simplicityDiscoverySearch
  3  * @namespace Performs a search against the Discovery Search Engine
  4  *
  5  * @example
  6  *   $('body').simplicityState();
  7  *   // Create all simplicityInputs widgets
  8  *   $('body')
  9  *     .simplicityState('mergeQueryParams')
 10  *     .simplicityHistory()
 11  *     .simplicityState('triggerChangeEvent')
 12  *     .simplicityPageSnapBack()
 13  *     <b>.simplicityDiscoverySearch({
 14  *       url: 'search_controller.php'
 15  *     })</b>
 16  *     <b>.simplicityDiscoverySearch('search');</b>
 17  */
 18 (function ($) {
 19   $.widget("ui.simplicityDiscoverySearch", $.ui.simplicityWidget, {
 20     /**
 21      * Widget options.
 22      *
 23      * <dl>
 24      *   <dt>stateElement</dt>
 25      *   <dd>
 26      *     The location of the simplicityState widget. Defaults to <code>'body'</code>.
 27      *   </dd>
 28      *   <dt>searchOnStateChange</dt>
 29      *   <dd>
 30      *     When <code>true</code> triggers a search everytime the state changes. Defaults to <code>true</code>.
 31      *   </dd>
 32      *   <dt>initialSearchResponse</dt>
 33      *   <dd>
 34      *     The initial search response. Used when a page has a server-side search triggered during load.
 35      *     Defaults to <code>{}</code>.
 36      *   </dd>
 37      *   <dt>dataType</dt>
 38      *   <dd>
 39      *     Defaults to 'json', change to 'jsonp' to allow for cross domain search controller support.
 40      *     This requires that your search controller output valid JSONP.
 41      *   <dt>debug</dt>
 42      *   <dd>
 43      *     Enable logging of key events to <code>console.log</code>. Defaults to <code>false</code>.
 44      *   </dd>
 45      * </dl>
 46      * @name $.ui.simplicityDiscoverySearch.options
 47      */
 48     options : {
 49       stateElement: 'body',
 50       searchOnStateChange: true,
 51       initialSearchResponse: {},
 52       dataType: 'json',
 53       debug: false,
 54       triggerFacetCountEvent: false,
 55       triggerResultSetEvent: false,
 56       getPayloadParam: ''
 57     },
 58     _create : function () {
 59       this._addClass('ui-simplicity-discovery-search');
 60       this.searchResponse(JSON.stringify(this.options.initialSearchResponse), false);
 61       this._bind(this.options.stateElement, 'simplicityStateChange', this._stateChangeHandler);
 62       this._ajaxHelperSearch = $.simplicityAjaxHelper(this);
 63       this._ajaxHelperQuery = $.simplicityAjaxHelper(this);
 64     },
 65     _stateChangeHandler: function (evt, state) {
 66       if (this.options.searchOnStateChange) {
 67         this.search(state);
 68       }
 69     },
 70     /**
 71      * Runs a search. The search happens asynchronously and will trigger multiple events.
 72      * <ul>
 73      *   <li>simplicitySearchResponse (original response)</li>
 74      *   <li>simplicityFacetCounts</li>
 75      *   <li>simplicityResultSet</li>
 76      * </ul>
 77      *
 78      * @param searchState
 79      *   The search state as a JSON compatible <code>Object</code>. This is an optional parameter,
 80      *   if not specified the search state is obtained from the <code>simplicityState</code> widget.
 81      *
 82      * @name $.ui.simplicityDiscoverySearch.search
 83      * @function
 84      *
 85      * @example
 86      *   // Bind to a search button
 87      *   $('#searchButton').click(function (evt) {
 88      *     $('body').simplicityDiscoverySearch('search');
 89      *   });
 90      */
 91     search: function (searchState) {
 92       if (arguments.length === 0) {
 93         searchState = $(this.options.stateElement).simplicityState('state');
 94       }
 95       if (this.options.debug) {
 96         console.log('Searching for ', this.element, 'with state', searchState);
 97       }
 98       this._ajaxHelperSearch.ajax({
 99         url: this.options.url,
100         type: 'GET',
101         contentType: 'application/json',
102         data: searchState,
103         dataType: this.options.dataType,
104         cache: false,
105         debug: this.options.debug,
106         error: this._errorHandler,
107         success: function (data, textStatus, xhr) {
108             if (this.options.debug) {
109               console.log('Search success for', this.element, 'with response', data);
110             }
111             this.searchResponse(data);
112           }
113       });
114     },
115     query: function (jsonString) {
116       var data = jsonString;
117       if (this.options.dataType === 'jsonp') {
118         try {
119           if (this.options.getPayloadParam !== '') {
120             data = {};
121             data[this.options.getPayloadParam] = JSON.stringify(JSON.parse(jsonString));
122           } else {
123             data = JSON.parse(jsonString);
124           }
125         }
126         catch (e) {
127           this.searchResponse({error: true, message: "Invalid request JSON"});
128           return;
129         }
130       }
131       this._ajaxHelperQuery.ajax({
132         url: this.options.url,
133         type: 'POST',
134         contentType: 'application/json',
135         data: data,
136         dataType: this.options.dataType,
137         cache: false,
138         debug: this.options.debug,
139         error: this._errorHandler,
140         success: function (data, textStatus, xhr) {
141             if (this.options.debug) {
142               console.log('Search success for', this.element, 'with response', data);
143             }
144             this.searchResponse({
145               _discovery: {
146                 request: JSON.parse(jsonString),
147                 response: data
148               }
149             });
150           }
151       });
152     },
153     _errorHandler: function (xhr, textStatus, message) {
154       if (this.options.debug) {
155         console.log('Search error for', this.element, 'textStatus:', textStatus, 'arguments', arguments);
156       }
157       this.searchResponse({
158         error: true,
159         xhr: xhr,
160         status: xhr.status,
161         statusText: textStatus,
162         message: message
163       });
164     },
165     /**
166      * Get or set (and process) the search response. Called with no arguments to retrieve the
167      * last response. Called with arguments to process the current response (called by
168      * <code>search</code> on success or failure).
169      *
170      * In processing mode, this method triggers a <code>simplicitySearchResponse</code>
171      * event and then calls <code>facetCounts</code> and <code>resultSet</code>.
172      *
173      *
174      * @param searchResponse
175      *   The search response to process.
176      * @param triggerEvent
177      *   Optional parameter, set it to <code>false</code> to prevent triggering of events.
178      *   Defaults to <code>true</code> if not specified.
179      *
180      * @example
181      *   // Get the last search response
182      *   var response = $('body').simplicityDiscoverySearch('searchResponse');
183      *
184      * @example
185      *   // Process a fake search response for testing purposes
186      *   var response = {
187      *     // Fake response
188      *   };
189      *   $('body').simplicityDiscoverySearch('searchResponse', response);
190      *
191      * @name $.ui.simplicityDiscoverySearch.searchResponse
192      * @function
193      */
194     searchResponse: function (searchResponse, triggerEvent) {
195       if (arguments.length === 0) {
196         return JSON.parse(this._searchResponse);
197       }
198       var discoveryResponse = searchResponse._discovery || {};
199       var response = discoveryResponse.response || {};
200       if (this.options.triggerFacetCountEvent) {
201         searchResponse.drillDown = this.extractFacetCounts(response);
202       }
203       searchResponse.resultSet = this.extractResultSet(response);
204       this._searchResponse = JSON.stringify(searchResponse);
205       if (triggerEvent !== false) {
206         this.element.triggerHandler('simplicitySearchResponse',
207           [JSON.parse(this._searchResponse)]);
208       }
209       if (this.options.triggerFacetCountEvent) {
210         this.facetCounts(searchResponse.drillDown, triggerEvent);
211       }
212       if (this.options.triggerResultSetEvent) {
213         this.resultSet(searchResponse.resultSet, triggerEvent);
214       }
215       if (triggerEvent !== false) {
216         this.element.triggerHandler('simplicitySearchResponseHandled');
217       }
218     },
219     /**
220      * Get the or set the facet counts. Called with no arguments for get behavior.
221      * Triggers a <code>simplicityFacetCounts</code> event on set.
222      *
223      * @param facetCounts
224      *   The facet counts to store.
225      * @param triggerEvent
226      *   Optional parameter, set it to <code>false</code> to prevent triggering of events.
227      *   Defaults to <code>true</code> if not specified.
228      *
229      * @name $.ui.simplicityDiscoverySearch.facetCounts
230      * @function
231      */
232     facetCounts: function (facetCounts, triggerEvent) {
233       if (arguments.length === 0) {
234         return JSON.parse(this._facetCounts);
235       }
236       this._facetCounts = JSON.stringify(facetCounts);
237       if (triggerEvent !== false) {
238         this.element.triggerHandler('simplicityFacetCounts', [JSON.parse(this._facetCounts)]);
239       }
240     },
241     /**
242      * Get the or set the resultset. Called with no arguments for get behavior.
243      * Triggers a <code>simplicityResultSet</code> event on set.
244      *
245      * @param resultSet
246      *   The resultset to store.
247      * @param triggerEvent
248      *   Optional parameter, set it to <code>false</code> to prevent triggering of events.
249      *   Defaults to <code>true</code> if not specified.
250      *
251      * @name $.ui.simplicityDiscoverySearch.resultSet
252      * @function
253      */
254     resultSet: function (resultSet, triggerEvent) {
255       if (arguments.length === 0) {
256         return JSON.parse(this._resultSet);
257       }
258       this._resultSet = JSON.stringify(resultSet);
259       if (triggerEvent !== false) {
260         this.element.triggerHandler('simplicityResultSet', [JSON.parse(this._resultSet)]);
261       }
262     },
263     /**
264      * Converts the search response to a facet counts object using
265      * the drillDown response.
266      *
267      * @example
268      *   Input
269      *   {
270      *     drillDown: [
271      *       {
272      *         dimension: shape,
273      *         ids: ['a', 'b', 'c'],
274      *         exactCounts: [1, 3, 4],
275      *         fuzzyCounts: [2, 6, 8]
276      *       }
277      *     ]
278      *   }
279      *
280      * @example
281      *   Output
282      *   {
283      *     shape: {
284      *       exact: {'a': 1, 'b': 3, 'c': 4},
285      *       fuzzy: {'a': 2, 'b': 6, 'c': 8}
286      *     }
287      *   }
288      *
289      * @param discoveryResponse
290      * @name $.ui.simplicityDiscoverySearch.extractFacetCounts
291      * @function
292      */
293     extractFacetCounts: function (discoveryResponse) {
294       var facetCounts = {};
295       if ($.isArray(discoveryResponse.drillDown)) {
296         $.each(discoveryResponse.drillDown, function (idx, elem) {
297           var dimFacetCounts = facetCounts[elem.dimension] || {};
298           facetCounts[elem.dimension] = dimFacetCounts;
299           var exactCounts = dimFacetCounts.exact || {};
300           dimFacetCounts.exact = exactCounts;
301           var fuzzyCounts = dimFacetCounts.fuzzy || {};
302           dimFacetCounts.fuzzy = fuzzyCounts;
303           $.each(elem.ids, function (idx, id) {
304             exactCounts[id] = elem.exactCounts[idx];
305             fuzzyCounts[id] = elem.fuzzyCounts[idx];
306           });
307         });
308       }
309       return facetCounts;
310     },
311     /**
312      * Converts the search response to a resultset object.
313      * @example
314      *   Input
315      *   {
316      *     startIndex: 0
317      *     pageSize: 10,
318      *     currentPageSize: 10,
319      *     exactSize: 2,
320      *     totalSize: 120,
321      *     datasetSize: 55000,
322      *     itemIds: ['a', 'b', 'c'],
323      *     relevanceValues: [1.0, 1.0, 0.9],
324      *     exactMatches: [true, true, false],
325      *     properties: [
326      *       {title: 'aaa'},
327      *       {title: 'bbb'},
328      *       {title: 'ccc'}
329      *     ],
330      *     highlighting: [
331      *       {},
332      *       {'title': '[b]bbb[/b]'},
333      *       {}
334      *     ]
335      *   }
336      *
337      * @example
338      *   Output
339      *   <pre>
340      *   {
341      *     startIndex: 0,
342      *     pageSize: 10,
343      *     exactSize: 2,
344      *     totalSize: 120,
345      *     datasetSize: 55000,
346      *     numRows: 10,
347      *     rows: [
348      *       {
349      *         id: 'a',
350      *         exact: true,
351      *         score: 1.0
352      *         properties: {
353      *           title: 'aaa'
354      *         },
355      *         highlighting: {}
356      *       },
357      *       {
358      *         id: 'b',
359      *         exact: true,
360      *         score: 1.0
361      *         properties: {
362      *           title: 'bbb'
363      *         },
364      *         highlighting: {
365      *           title: '[b]bbb[/b]'
366      *         }
367      *       },
368      *       {
369      *         id: 'c',
370      *         exact: false,
371      *         score: 0.9
372      *         properties: {
373      *           title: 'ccc'
374      *         },
375      *         highlighting: {}
376      *       }
377      *     ]
378      *   }
379      *   </pre>
380      *
381      * @param discoveryResponse
382      * @name $.ui.simplicityDiscoverySearch.extractResultSet
383      * @function
384      */
385     extractResultSet: function (discoveryResponse) {
386       var resultSet = {
387         startIndex: discoveryResponse.startIndex || 0,
388         pageSize: discoveryResponse.pageSize || 0,
389         exactSize: discoveryResponse.exactSize || 0,
390         totalSize: discoveryResponse.totalSize || 0,
391         datasetSize: discoveryResponse.datasetSize || 0
392       };
393       var rows = [];
394       if ('undefined' !== typeof discoveryResponse.itemIds) {
395         var itemIds = discoveryResponse.itemIds;
396         var exactMatches = discoveryResponse.exactMatches;
397         var relevanceValues = discoveryResponse.relevanceValues;
398         var properties = discoveryResponse.properties;
399         var highlighting = discoveryResponse.highlighting;
400         if (!$.isArray(properties)) {
401           properties = null;
402         }
403         if (!$.isArray(highlighting)) {
404           highlighting = null;
405         }
406         $.each(itemIds, function (idx, itemId) {
407           row = {
408             id: itemId,
409             exact: exactMatches[idx],
410             score: relevanceValues[idx]
411           };
412           if (properties !== null) {
413             row.properties = properties[idx];
414           }
415           if (highlighting !== null) {
416             row.highlighting = highlighting[idx];
417           }
418           rows.push(row);
419         });
420       }
421       resultSet.numRows = rows.length;
422       resultSet.rows = rows;
423       return resultSet;
424     },
425     searchStats: function () {
426       return this._ajaxHelperSearch.getStats();
427     },
428     queryStats: function () {
429       return this._ajaxHelperQuery.getStats();
430     }
431   });
432   $.fn.simplicityDiscoverySearchItemEnumerator = function (searchResponse, callback) {
433     var discovery = searchResponse._discovery || {};
434     var discoveryResponse = discovery.response || {};
435     if ('undefined' !== typeof discoveryResponse.itemIds) {
436       var startIndex = discoveryResponse.startIndex;
437       var itemIds = discoveryResponse.itemIds;
438       var exactMatches = discoveryResponse.exactMatches;
439       var relevanceValues = discoveryResponse.relevanceValues;
440       var properties = discoveryResponse.properties;
441       var highlighting = discoveryResponse.highlighting;
442       if (!$.isArray(properties)) {
443         properties = null;
444       }
445       if (!$.isArray(highlighting)) {
446         highlighting = null;
447       }
448       $.each(itemIds, function (idx, itemId) {
449         row = {
450           id: itemId,
451           exact: exactMatches[idx],
452           score: relevanceValues[idx],
453           resultsIndex0: idx + startIndex,
454           index0: idx,
455           index1: idx + 1
456         };
457         if (properties !== null) {
458           row.properties = properties[idx];
459         }
460         if (highlighting !== null) {
461           row.highlighting = highlighting[idx];
462         }
463         callback(idx, row);
464       });
465     }
466   };
