1 /**
  2  * @name $.ui.simplicityDiscoverySearch
  3  * @namespace Performs a search against the Discovery Search Engine
  4  *
  5  * @example
  6  *   $('body').simplicityState();
  7  *   // Create all simplicityInputs widgets
  8  *   $('body')
  9  *     .simplicityState('mergeQueryParams')
 10  *     .simplicityHistory()
 11  *     .simplicityState('triggerChangeEvent')
 12  *     .simplicityPageSnapBack()
 13  *     <b>.simplicityDiscoverySearch({
 14  *       url: 'search_controller.php'
 15  *     })</b>
 16  *     <b>.simplicityDiscoverySearch('search');</b>
 17  */
 18 (function ($) {
 19   $.widget("ui.simplicityDiscoverySearch", $.ui.simplicityWidget, {
 20     /**
 21      * Widget options.
 22      *
 23      * <dl>
 24      *   <dt>stateElement</dt>
 25      *   <dd>
 26      *     The location of the simplicityState widget. Defaults to <code>'body'</code>.
 27      *   </dd>
 28      *   <dt>searchOnStateChange</dt>
 29      *   <dd>
 30      *     When <code>true</code> triggers a search everytime the state changes. Defaults to <code>true</code>.
 31      *   </dd>
 32      *   <dt>initialSearchResponse</dt>
 33      *   <dd>
 34      *     The initial search response. Used when a page has a server-side search triggered during load.
 35      *     Defaults to <code>{}</code>.
 36      *   </dd>
 37      *   <dt>dataType</dt>
 38      *   <dd>
 39      *     Defaults to 'json', change to 'jsonp' to allow for cross domain search controller support.
 40      *     This requires that your search controller output valid JSONP.
 41      *   <dt>debug</dt>
 42      *   <dd>
 43      *     Enable logging of key events to <code>console.log</code>. Defaults to <code>false</code>.
 44      *   </dd>
 45      *   <dt>profile</dt>
 46      *   <dd>
 47      *     When <code>true</code> requests query profile data from the engine. The profile data
 48      *     can be used to analyze engine query response times. Defaults to <code>false</code>.
 49      *   </dd>
 50      *   <dt>triggerFacetCountEvent</dt>
 51      *   <dd>
 52      *     When <code>true</code> enables triggering of facet count events. Defaults to <code>false</code>.
 53      *   </dd>
 54      *   <dt>triggerResultSetEvent</dt>
 55      *   <dd>
 56      *     When <code>true</code> enables triggering of the simplicitySearchResultSet event. Defaults to <code>false</code>.
 57      *   </dd>
 58      * </dl>
 59      * @name $.ui.simplicityDiscoverySearch.options
 60      */
 61     options : {
 62       stateElement: 'body',
 63       searchOnStateChange: true,
 64       initialSearchResponse: {},
 65       dataType: 'json',
 66       debug: false,
 67       triggerFacetCountEvent: false,
 68       triggerResultSetEvent: false,
 69       profile: false,
 70       getPayloadParam: ''
 71     },
 72     _create : function () {
 73       this._addClass('ui-simplicity-discovery-search');
 74       this.searchResponse(JSON.stringify(this.options.initialSearchResponse), false);
 75       this._bind(this.options.stateElement, 'simplicityStateChange', this._stateChangeHandler);
 76       this._ajaxHelperSearch = $.simplicityAjaxHelper(this);
 77       this._ajaxHelperQuery = $.simplicityAjaxHelper(this);
 78     },
 79     _stateChangeHandler: function (evt, state) {
 80       if (this.options.searchOnStateChange) {
 81         this.search(state);
 82       }
 83     },
 84     /**
 85      * Runs a search. The search happens asynchronously and will trigger multiple events.
 86      * <ul>
 87      *   <li>simplicitySearchResponse (original response)</li>
 88      *   <li>ajaxSetup</li>
 89      *   <li>ajaxSuccess</li>
 90      *   <li>ajaxError</li>
 91      *   <li>simplicityFacetCounts</li>
 92      *   <li>simplicityResultSet</li>
 93      * </ul>
 94      *
 95      * @param searchState
 96      *   The search state as a JSON compatible <code>Object</code>. This is an optional parameter,
 97      *   if not specified the search state is obtained from the <code>simplicityState</code> widget.
 98      *
 99      * @name $.ui.simplicityDiscoverySearch.search
100      * @function
101      *
102      * @example
103      *   // Bind to a search button
104      *   $('#searchButton').click(function (evt) {
105      *     $('body').simplicityDiscoverySearch('search');
106      *   });
107      */
108     search: function (searchState) {
109       if (arguments.length === 0) {
110         searchState = $(this.options.stateElement).simplicityState('state');
111       }
112       if (this.options.debug) {
113         console.log('Searching for ', this.element, 'with state', searchState);
114       }
115       var ajaxOptions = {
116           url: this.options.url,
117           type: 'GET',
118           contentType: 'application/json',
119           data: searchState,
120           dataType: this.options.dataType,
121           cache: false,
122           beforeSend: $.proxy(this._beforeSend, this),
123           debug: this.options.debug,
124           error: this._errorHandler,
125           success: function (data, textStatus, xhr) {
126             if (this.options.debug) {
127               console.log('Search success for', this.element, 'with response', data);
128             }
129             this.element.triggerHandler('ajaxSuccess', data, textStatus, xhr);
130             this._addHeadersToResponse(data, xhr);
131             this.searchResponse(data);
132           }
133         };
134       this.element.triggerHandler('ajaxSetup', ajaxOptions);
135       this._ajaxHelperSearch.ajax(ajaxOptions);
136     },
137     query: function (jsonString) {
138       var data = jsonString;
139       if (this.options.dataType === 'jsonp') {
140         try {
141           if (this.options.getPayloadParam !== '') {
142             data = {};
143             data[this.options.getPayloadParam] = JSON.stringify(JSON.parse(jsonString));
144           } else {
145             data = JSON.parse(jsonString);
146           }
147         }
148         catch (e) {
149           this.searchResponse({error: true, message: "Invalid request JSON"});
150           return;
151         }
152       }
153       var ajaxOptions = {
154           url: this.options.url,
155           type: 'POST',
156           contentType: 'application/json',
157           data: data,
158           dataType: this.options.dataType,
159           cache: false,
160           beforeSend: $.proxy(this._beforeSend, this),
161           debug: this.options.debug,
162           error: this._errorHandler,
163           success: function (data, textStatus, xhr) {
164               if (this.options.debug) {
165                 console.log('Search success for', this.element, 'with response', data);
166               }
167               this.element.triggerHandler('ajaxSuccess', data, textStatus, xhr);
168               var response = {
169                   _discovery: {
170                     request: JSON.parse(jsonString),
171                     response: data
172                   }
173                 };
174               this._addHeadersToResponse(response, xhr);
175               this.searchResponse(response);
176             }
177         };
178       this.element.triggerHandler('ajaxSetup', ajaxOptions);
179       this._ajaxHelperQuery.ajax(ajaxOptions);
180     },
181     _beforeSend: function (xhr) {
182       if (this.options.profile) {
183         xhr.setRequestHeader('X-Profile', 1);
184       }
185     },
186     _addHeadersToResponse: function (response, xhr) {
187       var profile = xhr.getResponseHeader('X-Profiling-Trace');
188       if (profile > '') {
189         response._discovery.profile = profile.substr(5).split(', >>>   ');
190       }
191     },
192     _errorHandler: function (xhr, textStatus, message) {
193       if (this.options.debug) {
194         console.log('Search error for', this.element, 'textStatus:', textStatus, 'arguments', arguments);
195       }
196       this.element.triggerHandler('ajaxError', xhr, textStatus, message);
197       this.searchResponse({
198         error: true,
199         xhr: xhr,
200         status: xhr.status,
201         statusText: textStatus,
202         message: message
203       });
204     },
205     /**
206      * Get or set (and process) the search response. Called with no arguments to retrieve the
207      * last response. Called with arguments to process the current response (called by
208      * <code>search</code> on success or failure).
209      *
210      * In processing mode, this method triggers <code>simplicitySearchResponse</code>
211      * and <code>simplicitySearchResponseHandled</code> events, then calls
212      * <code>facetCounts</code> and <code>resultSet</code>.
213      *
214      *
215      * @param searchResponse
216      *   The search response to process.
217      * @param triggerEvent
218      *   Optional parameter, set it to <code>false</code> to prevent triggering of events.
219      *   Defaults to <code>true</code> if not specified.
220      *
221      * @example
222      *   // Get the last search response
223      *   var response = $('body').simplicityDiscoverySearch('searchResponse');
224      *
225      * @example
226      *   // Process a fake search response for testing purposes
227      *   var response = {
228      *     // Fake response
229      *   };
230      *   $('body').simplicityDiscoverySearch('searchResponse', response);
231      *
232      * @name $.ui.simplicityDiscoverySearch.searchResponse
233      * @function
234      */
235     searchResponse: function (searchResponse, triggerEvent) {
236       if (arguments.length === 0) {
237         return JSON.parse(this._searchResponse);
238       }
239       var discoveryResponse = searchResponse._discovery || {};
240       var response = discoveryResponse.response || {};
241       if (this.options.triggerFacetCountEvent) {
242         searchResponse.drillDown = this.extractFacetCounts(response);
243       }
244       searchResponse.resultSet = this.extractResultSet(response);
245       this._searchResponse = JSON.stringify(searchResponse);
246       if (triggerEvent !== false) {
247         this.element.triggerHandler('simplicitySearchResponse',
248           [JSON.parse(this._searchResponse)]);
249       }
250       if (this.options.triggerFacetCountEvent) {
251         this.facetCounts(searchResponse.drillDown, triggerEvent);
252       }
253       if (this.options.triggerResultSetEvent) {
254         this.resultSet(searchResponse.resultSet, triggerEvent);
255       }
256       if (triggerEvent !== false) {
257         this.element.triggerHandler('simplicitySearchResponseHandled');
258       }
259     },
260     /**
261      * Get the or set the facet counts. Called with no arguments for get behavior.
262      * Triggers a <code>simplicityFacetCounts</code> event on set.
263      *
264      * @param facetCounts
265      *   The facet counts to store.
266      * @param triggerEvent
267      *   Optional parameter, set it to <code>false</code> to prevent triggering of events.
268      *   Defaults to <code>true</code> if not specified.
269      *
270      * @name $.ui.simplicityDiscoverySearch.facetCounts
271      * @function
272      */
273     facetCounts: function (facetCounts, triggerEvent) {
274       if (arguments.length === 0) {
275         return JSON.parse(this._facetCounts);
276       }
277       this._facetCounts = JSON.stringify(facetCounts);
278       if (triggerEvent !== false) {
279         this.element.triggerHandler('simplicityFacetCounts', [JSON.parse(this._facetCounts)]);
280       }
281     },
282     /**
283      * Get the or set the resultset. Called with no arguments for get behavior.
284      * Triggers a <code>simplicityResultSet</code> event on set.
285      *
286      * @param resultSet
287      *   The resultset to store.
288      * @param triggerEvent
289      *   Optional parameter, set it to <code>false</code> to prevent triggering of events.
290      *   Defaults to <code>true</code> if not specified.
291      *
292      * @name $.ui.simplicityDiscoverySearch.resultSet
293      * @function
294      */
295     resultSet: function (resultSet, triggerEvent) {
296       if (arguments.length === 0) {
297         return JSON.parse(this._resultSet);
298       }
299       this._resultSet = JSON.stringify(resultSet);
300       if (triggerEvent !== false) {
301         this.element.triggerHandler('simplicityResultSet', [JSON.parse(this._resultSet)]);
302       }
303     },
304     /**
305      * Converts the search response to a facet counts object using
306      * the drillDown response.
307      *
308      * @example
309      *   Input
310      *   {
311      *     drillDown: [
312      *       {
313      *         dimension: shape,
314      *         ids: ['a', 'b', 'c'],
315      *         exactCounts: [1, 3, 4],
316      *         fuzzyCounts: [2, 6, 8]
317      *       }
318      *     ]
319      *   }
320      *
321      * @example
322      *   Output
323      *   {
324      *     shape: {
325      *       exact: {'a': 1, 'b': 3, 'c': 4},
326      *       fuzzy: {'a': 2, 'b': 6, 'c': 8}
327      *     }
328      *   }
329      *
330      * @param discoveryResponse
331      * @name $.ui.simplicityDiscoverySearch.extractFacetCounts
332      * @function
333      */
334     extractFacetCounts: function (discoveryResponse) {
335       var facetCounts = {};
336       if ($.isArray(discoveryResponse.drillDown)) {
337         $.each(discoveryResponse.drillDown, function (idx, elem) {
338           var dimFacetCounts = facetCounts[elem.dimension] || {};
339           facetCounts[elem.dimension] = dimFacetCounts;
340           var exactCounts = dimFacetCounts.exact || {};
341           dimFacetCounts.exact = exactCounts;
342           var fuzzyCounts = dimFacetCounts.fuzzy || {};
343           dimFacetCounts.fuzzy = fuzzyCounts;
344           $.each(elem.ids, function (idx, id) {
345             exactCounts[id] = elem.exactCounts[idx];
346             fuzzyCounts[id] = elem.fuzzyCounts[idx];
347           });
348         });
349       }
350       return facetCounts;
351     },
352     /**
353      * Converts the search response to a resultset object.
354      * @example
355      *   Input
356      *   {
357      *     startIndex: 0
358      *     pageSize: 10,
359      *     currentPageSize: 10,
360      *     exactSize: 2,
361      *     totalSize: 120,
362      *     datasetSize: 55000,
363      *     itemIds: ['a', 'b', 'c'],
364      *     relevanceValues: [1.0, 1.0, 0.9],
365      *     exactMatches: [true, true, false],
366      *     properties: [
367      *       {title: 'aaa'},
368      *       {title: 'bbb'},
369      *       {title: 'ccc'}
370      *     ],
371      *     highlighting: [
372      *       {},
373      *       {'title': '[b]bbb[/b]'},
374      *       {}
375      *     ]
376      *   }
377      *
378      * @example
379      *   Output
380      *   <pre>
381      *   {
382      *     startIndex: 0,
383      *     pageSize: 10,
384      *     exactSize: 2,
385      *     totalSize: 120,
386      *     datasetSize: 55000,
387      *     numRows: 10,
388      *     rows: [
389      *       {
390      *         id: 'a',
391      *         exact: true,
392      *         score: 1.0
393      *         properties: {
394      *           title: 'aaa'
395      *         },
396      *         highlighting: {}
397      *       },
398      *       {
399      *         id: 'b',
400      *         exact: true,
401      *         score: 1.0
402      *         properties: {
403      *           title: 'bbb'
404      *         },
405      *         highlighting: {
406      *           title: '[b]bbb[/b]'
407      *         }
408      *       },
409      *       {
410      *         id: 'c',
411      *         exact: false,
412      *         score: 0.9
413      *         properties: {
414      *           title: 'ccc'
415      *         },
416      *         highlighting: {}
417      *       }
418      *     ]
419      *   }
420      *   </pre>
421      *
422      * @param discoveryResponse
423      * @name $.ui.simplicityDiscoverySearch.extractResultSet
424      * @function
425      */
426     extractResultSet: function (discoveryResponse) {
427       var resultSet = {
428         startIndex: discoveryResponse.startIndex || 0,
429         pageSize: discoveryResponse.pageSize || 0,
430         exactSize: discoveryResponse.exactSize || 0,
431         totalSize: discoveryResponse.totalSize || 0,
432         datasetSize: discoveryResponse.datasetSize || 0
433       };
434       var rows = [];
435       if ('undefined' !== typeof discoveryResponse.itemIds) {
436         var itemIds = discoveryResponse.itemIds;
437         var exactMatches = discoveryResponse.exactMatches;
438         var relevanceValues = discoveryResponse.relevanceValues;
439         var properties = discoveryResponse.properties;
440         var highlighting = discoveryResponse.highlighting;
441         if (!$.isArray(properties)) {
442           properties = null;
443         }
444         if (!$.isArray(highlighting)) {
445           highlighting = null;
446         }
447         $.each(itemIds, function (idx, itemId) {
448           row = {
449             id: itemId,
450             exact: exactMatches[idx],
451             score: relevanceValues[idx]
452           };
453           if (properties !== null) {
454             row.properties = properties[idx];
455           }
456           if (highlighting !== null) {
457             row.highlighting = highlighting[idx];
458           }
459           rows.push(row);
460         });
461       }
462       resultSet.numRows = rows.length;
463       resultSet.rows = rows;
464       return resultSet;
465     },
466     searchStats: function () {
467       return this._ajaxHelperSearch.getStats();
468     },
469     queryStats: function () {
470       return this._ajaxHelperQuery.getStats();
471     }
472   });
473   $.fn.simplicityDiscoverySearchItemEnumerator = function (searchResponse, callback) {
474     var discovery = searchResponse._discovery || {};
475     var discoveryResponse = discovery.response || {};
476     if ('undefined' !== typeof discoveryResponse.itemIds) {
477       var startIndex = discoveryResponse.startIndex;
478       var itemIds = discoveryResponse.itemIds;
479       var exactMatches = discoveryResponse.exactMatches;
480       var relevanceValues = discoveryResponse.relevanceValues;
481       var properties = discoveryResponse.properties;
482       var highlighting = discoveryResponse.highlighting;
483       if (!$.isArray(properties)) {
484         properties = null;
485       }
486       if (!$.isArray(highlighting)) {
487         highlighting = null;
488       }
489       $.each(itemIds, function (idx, itemId) {
490         row = {
491           id: itemId,
492           exact: exactMatches[idx],
493           score: relevanceValues[idx],
494           resultsIndex0: idx + startIndex,
495           index0: idx,
496           index1: idx + 1
497         };
498         if (properties !== null) {
499           row.properties = properties[idx];
500         }
501         if (highlighting !== null) {
502           row.highlighting = highlighting[idx];
503         }
504         callback(idx, row);
505       });
506     }
507   };
