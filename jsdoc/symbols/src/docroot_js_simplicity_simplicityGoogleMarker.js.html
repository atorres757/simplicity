1 /**
  2  * A custom Google Maps Overlay View for a jQuery-based standard marker
  3  *
  4  * @class $.simplicityGoogleMarker
  5  * @extends google.maps.OverlayView
  6  * @param {jQuery} element jQuery element to use for the marker
  7  * @param {Dictionary} row simplicityGoogleMapResults ui.row data for the marker
  8  * @param {Dictionary} options Marker options
  9  * @property {jQuery} element The jQuery element to use as the marker.
 10  * @property {Dictionary} row simplicityGoogleMapResults ui.row data for the marker.
 11  * @property {Dictionary} options Marker options
 12  * <dl>
 13  *   <dt>position</dt>
 14  *   <dd>
 15  *     The latLng for the marker.
 16  *   </dd>
 17  *   <dt>anchor</dt>
 18  *   <dd>
 19  *     The anchor...to position the marker over the latLng point.
 20  *   </dd>
 21  *   <dt>draggable</dt>
 22  *   <dd>
 23  *     Indicates that the marker is draggble. Defaults to <code>false</code>.
 24  *   </dd>
 25  * </dl>
 26  */
 27 (function ($) {
 28   $.simplicityGoogleMarker = function (element, row, options) {
 29     this.element = $(element);
 30     this.row = row;
 31     this.options = options;
 32     this.latLng = options.position;
 33     this.element.addClass('ui-simplicity-google-marker');
 34     if (this.options.draggable) {
 35       this.element.draggable({
 36         stop: $.proxy(function (evt, ui) {
 37           this.setLatLngFromOffset(
 38             ui.position.left - ui.originalPosition.left, // dX
 39             ui.position.top - ui.originalPosition.top // dY
 40           );
 41         }, this)
 42       });
 43     }
 44     /**
 45      * Required implementation of <code>google.maps.OverlayView.onAdd</code>
 46      *
 47      * @name $.simplicityGoogleMarker.onAdd
 48      * @public
 49      */
 50     this.onAdd = function () {
 51       this.getPanes().overlayMouseTarget.appendChild(this.element[0]);
 52     };
 53     /**
 54      * Required implementation of <code>google.maps.OverlayView.onRemove</code>
 55      *
 56      * @name $.simplicityGoogleMarker.onRemove
 57      * @public
 58      */
 59     this.onRemove = function () {
 60       this.element.remove();
 61     };
 62     /**
 63      * Required implementation of <code>google.maps.OverlayView.draw</code>
 64      *
 65      * @name $.simplicityGoogleMarker.draw
 66      * @public
 67      */
 68     this.draw = function () {
 69       if ('undefined' !== typeof this.latLng) {
 70         this.latLngPt = this.getProjection().fromLatLngToDivPixel(this.latLng);
 71         var h = this.element.height();
 72        // default bottom center
 73         var a = this.options.anchor || new google.maps.Point(this.element.width() / 2, h);
 74         var p = new google.maps.Point(this.latLngPt.x - a.x, this.latLngPt.y - a.y - (this.row.index0 * h));
 75         this.element.css({
 76           top:  p.y + 'px',
 77           left: p.x + 'px',
 78           'z-Index': -this.row.index1
 79         });
 80       }
 81     };
 82     /**
 83      * Calculates new latLng for a marker based on it's new position relative
 84      * to it's previous position
 85      *
 86      * @name $.simplicityGoogleMarker.setLatLngFromOffset
 87      * @function
 88      * @param {double} dX Offset value for the X axis
 89      * @param {double} dY Offset value for the Y axis
 90      * @public
 91      */
 92     this.setLatLngFromOffset = function (dX, dY) {
 93       this.latLngPt = new google.maps.Point(dX + this.latLngPt.x, dY + this.latLngPt.y);
 94       this.latLng = this.getProjection().fromDivPixelToLatLng(this.latLngPt);
 95     };
 96     /**
 97      * Returns the marker's current latLng position
 98      *
 99      * @name $.simplicityGoogleMarker.getPosition
100      * @function
101      * @returns {google.maps.LatLng} The latLng of the marker
102      * @public
103      */
104     this.getPosition = function () {
105       return this.latLng;
106     };
107     /**
108      * Returns the marker's current latLng position
109      *
110      * @name $.simplicityGoogleMarker.setPosition
111      * @function
112      * @param {google.maps.LatLng} latLng The new latLng of the marker
113      * @public
114      */
115     this.setPosition = function (latLng) {
116       if ('undefined' !== typeof latLng) {
117         this.latLng = latLng;
118         this.draw();
119       }
120     };
121     /**
122      * Returns the marker's element
123      *
124      * @name $.simplicityGoogleMarker.getElement
125      * @function
126      * @returns {jQuery} The jQuery element used for the marker
127      * @public
128      */
129     this.getElement = function () {
130       return this.element;
131     };
132     /**
133      * Returns the marker's title
134      *
135      * @name $.simplicityGoogleMarker.setTitle
136      * @function
137      * @param {String} s The title to set
138      * @public
139      */
140     this.setTitle = function (s) {
141       this.element.attr('title', s);
142     };
143   };
144   $.simplicityGoogleMarker.prototype = new google.maps.OverlayView();
145 }(jQuery));
